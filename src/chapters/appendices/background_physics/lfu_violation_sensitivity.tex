\section{Sensitivity to LFU Violation}

Let
\begin{align}
\frac{g_e}{g_\mu}
\equiv
1+\Delta_{g_e/g_\mu}
\equiv x .
\end{align}

For small deviations from unity,
\begin{align}
x^2
\simeq
1+2(x-1).
\end{align}

Therefore,
\begin{align}
R_{e/\mu}
\simeq
\left(1+2\Delta_{g_e/g_\mu}\right)
\left[
\frac{
m_e(m_\pi^2-m_e^2)
}{
m_\mu(m_\pi^2-m_\mu^2)
}
\right]^2 .
\end{align}

Defining
\begin{align}
\Delta R_{e/\mu}
=
R_{e/\mu}
-
\left(R_{e/\mu}\right)_{\mathrm{theory}},
\end{align}

one obtains
\begin{align}
\frac{
\Delta R_{e/\mu}
}{
\left(R_{e/\mu}\right)_{\mathrm{theory}}
}
=
2\Delta_{g_e/g_\mu}.
\end{align}
